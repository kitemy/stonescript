// SPIT DODGING TRICK
func spitdodging()
 var incomingSpit = -1
 var incomingSpit2 = -1
 ?foe.id = ＂nagaraja＂
  ?foe.state = 31 | 
  ^(foe.state = 33 & foe.time = 0)
   ?incomingSpit > totaltime
    incomingSpit2 = totaltime + 25
   :
    incomingSpit = totaltime + 25
  ?foe.distance <= 10 &
  ^item.GetCooldown(＂mind＂) <= 0
   equipL mind
   equipR mask
  :?incomingSpit > totaltime
   ?incomingSpit - totaltime <= 6
    ?incomingSpit = totaltime + 6
     equip D ice crossbow
    :
     equipL triskel
     equipR sword
   :
    equip D ice crossbow
  :?foe.state = 105
   equip D ice crossbow
  :
   equip D ice crossbow
  ?incomingSpit = totaltime
  ^&incomingSpit2 > incomingSpit
   incomingSpit = incomingSpit2
