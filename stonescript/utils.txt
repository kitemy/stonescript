    ?item.GetCooldown(＂dash＂) <= 0
      ?foe.count >= 4 | (foe = boss & foe.buffs.string ! ＂mirror＂)

    equipL trisk

func moonjuggling(s1, s2)
    ?moon % 4 = 0
      equipR @s2@
      equipL @s1@
    :?moon % 4 = 1
      equipR moon
      equipL @s1@
    :?moon % 4 = 2
      equipR @s2@
      equipL @s1@
    :
      equipR @s2@
      equipL moon
    moon++

func CanUse(item)
  ?item.CanActivate() & 
  ^(item.GetCooldown(item) <= 0 
  ^| item.right = item
  ^ & item.right.state = 2)
    return true
  return false

func binSearchEnemy(min, max, foe_cnt)
  ?min = max
    return min
  var mid = (min + max)/2
  ?foe.GetCount(mid) < foe_cnt
    return binSearchEnemy(mid + 1, max, foe_cnt)
  :
    return binSearchEnemy(min, mid, foe_cnt)

func getFoeDistance(foe_cnt)
  return binSearchEnemy(0, 42, foe_cnt)

func get_berserk_pot_duration()
  ?buffs.string
    for buff : string.Split(buffs.string, ",")
      var info = string.Split(buff, ":")
      ?info[1] = "berserk" & info[1] ! "buff_staff_berserk"
        return int.Parse(info[3])
  return 0

func infReduction()
  ?!infTime
    rec = 0
  :?infTime
    ?!rec
      rec = infTime
    :?(rec + 2 - infTime) % 60 = 0
      equipL tri
      equipR tower
      return true 
  return false

    loops++
    old_t = item.GetTreasureCount()
    enduse = 0
    moondialTimer = 0
    swordtime = 0
    moon = 0
